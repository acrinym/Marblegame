That is a very valid critique, Justin. You are absolutely right—building a track out of a chain of static circles (the "beads" or "small pieces" you saw) is a quick-and-dirty way to prototype physics, but it results in a bumpy ride and looks terrible—completely failing to capture the smooth, "flow" aesthetic of Schauberger's designs or the mechanical precision of DaVinci.

The issue is that `Matter.Render` (the built-in renderer currently being used) is primarily a **debug** renderer. It shows the underlying physics bodies (the "colliders") rather than a beautiful graphic. Since the previous code built curves out of tiny circles, you saw tiny circles.

To fix this and get closer to your vision, we need to do two things:

1. **Fix the Physics Construction:** Instead of "beads," we will build the track walls out of **rotated rectangular segments**. This creates a much smoother rolling surface (like a polygon) and looks like a solid rail when rendered.
2. **Apply Textures:** We will use the textures you already have (like `wood.jpg`) so the track looks like material, not math.

Here is the verbose, corrected code for `CurvedTrack.ts` to generate smooth rails, and an updated `PhysicsEngine.tsx` to ensure the rendering handles these textures properly.

### 1. `client/src/components/contraptions/CurvedTrack.ts`

This replaces the "bead chain" with a "tangential rectangle chain." This ensures the marble rolls on a flat surface that turns incrementally, which is much smoother and allows us to stretch a texture across the segments.

```typescript
import Matter from "matter-js";

export interface CurvedTrack {
  bodies: Matter.Body[];
  id: string;
  startAngle: number;
  endAngle: number;
  radius: number;
}

/**
 * Creates a smooth curved track using rectangular segments rather than circles.
 * This provides a smoother rolling surface and looks more like a continuous rail.
 */
export function createCurvedTrack(
  world: Matter.World,
  centerX: number,
  centerY: number,
  id: string,
  radius: number = 100,
  startAngle: number = 0,
  endAngle: number = Math.PI / 2,
  trackWidth: number = 20 // Slightly wider to accommodate the marble comfortably
): CurvedTrack {
  const bodies: Matter.Body[] = [];
  
  // Determine direction of curve to ensure segments align correctly
  const totalAngle = endAngle - startAngle;
  const isClockwise = totalAngle > 0;
  
  // Calculate segment count based on radius to ensure smoothness
  // Larger radius = more segments needed to look smooth
  const segmentLength = 15; // Length of each rectangular segment
  const arcLength = Math.abs(totalAngle) * radius;
  const segmentCount = Math.ceil(arcLength / segmentLength);
  const angleStep = totalAngle / segmentCount;

  // We are building a "channel" track, so we need two parallel walls.
  // Wall 1: Outer Rail
  // Wall 2: Inner Rail
  const wallThickness = 6;
  const outerRadius = radius + trackWidth / 2;
  const innerRadius = radius - trackWidth / 2;

  // Helper to create a rail ring
  const createRail = (railRadius: number, labelSuffix: string) => {
    for (let i = 0; i < segmentCount; i++) {
      // Calculate the angle for the center of this segment
      const currentAngle = startAngle + (i * angleStep) + (angleStep / 2);
      
      // Position is based on the center of the segment arc
      const x = centerX + railRadius * Math.cos(currentAngle);
      const y = centerY + railRadius * Math.sin(currentAngle);
      
      // Rotation: The rectangle must be tangent to the circle at this angle.
      // The tangent is perpendicular to the radius (angle + 90 degrees).
      const rotation = currentAngle + (Math.PI / 2);

      // Create the segment body
      // We use a slight overlap (length + 1) to prevent "cracks" between segments
      const segment = Matter.Bodies.rectangle(x, y, segmentLength + 2, wallThickness, {
        isStatic: true,
        angle: rotation,
        label: `curved-track-${labelSuffix}-${id}-${i}`,
        friction: 0.05, // Low friction for smooth rolling
        restitution: 0.2, // Low bounce
        render: { 
          // Use the wood texture. 
          // Note: Textures in Matter.js rotate with the body, which is perfect here.
          sprite: {
            texture: '/textures/wood.jpg',
            xScale: (segmentLength + 2) / 512, // Scale texture to fit segment size (assuming approx 512px texture)
            yScale: wallThickness / 512
          }
        },
        // Rounding the edges slightly helps the marble roll over seams smoothly
        chamfer: { radius: 2 } 
      });
      
      bodies.push(segment);
    }
  };

  // Generate both rails
  createRail(outerRadius, "outer");
  createRail(innerRadius, "inner");
  
  Matter.Composite.add(world, bodies);
  
  return {
    bodies,
    id,
    startAngle,
    endAngle,
    radius,
  };
}

// These helper functions remain largely the same but use the new core generator
export function createQuarterPipe(
  world: Matter.World,
  x: number,
  y: number,
  id: string,
  direction: "ne" | "nw" | "se" | "sw" = "se",
  radius: number = 80
): CurvedTrack {
  let startAngle: number;
  let endAngle: number;
  let centerX = x;
  let centerY = y;
  
  // Adjust angles for canvas coordinate system (y increases downwards)
  switch (direction) {
    case "ne": // Top-Right quadrant
      startAngle = Math.PI;
      endAngle = 3 * Math.PI / 2; // 180 to 270
      centerX = x;
      centerY = y + radius;
      break;
    case "nw": // Top-Left quadrant
      startAngle = 3 * Math.PI / 2; // 270 to 360 (0)
      endAngle = 2 * Math.PI;
      centerX = x;
      centerY = y + radius;
      break;
    case "se": // Bottom-Right quadrant
      startAngle = Math.PI / 2;
      endAngle = Math.PI; // 90 to 180
      centerX = x;
      centerY = y - radius;
      break;
    case "sw": // Bottom-Left quadrant
    default:
      startAngle = 0;
      endAngle = Math.PI / 2; // 0 to 90
      centerX = x;
      centerY = y - radius;
      break;
  }
  
  return createCurvedTrack(world, centerX, centerY, id, radius, startAngle, endAngle);
}

export function createHalfPipe(
  world: Matter.World,
  x: number,
  y: number,
  id: string,
  orientation: "top" | "bottom" | "left" | "right" = "bottom",
  radius: number = 60
): CurvedTrack {
  let startAngle: number;
  let endAngle: number;
  
  switch (orientation) {
    case "top":
      startAngle = Math.PI;
      endAngle = 2 * Math.PI;
      break;
    case "bottom":
      startAngle = 0;
      endAngle = Math.PI;
      break;
    case "left":
      startAngle = -Math.PI / 2;
      endAngle = Math.PI / 2;
      break;
    case "right":
    default:
      startAngle = Math.PI / 2;
      endAngle = 3 * Math.PI / 2;
      break;
  }
  
  return createCurvedTrack(world, x, y, id, radius, startAngle, endAngle);
}
```

```typescript
    const colors: MarbleColor[] = ["red", "orange", "yellow", "green", "blue", "purple"];
    const binWidth = 80;
    const spacing = 120;
    const startX = (worldWidth - (colors.length - 1) * spacing) / 2;
    const y = worldHeight - 100;

    colors.forEach((color, index) => {
      const x = startX + index * spacing;
      const bin = createSingleExitBin(x, y, `exit-${color}`, color);
      bins.push(bin);
      exitBinsRef.current.set(`exit-${color}`, { body: bin, color });
    });

    return bins;
  };

  // The drop logic remains standard
  const dropMarbleInternal = (color: MarbleColor, marbleId: string, funnelIndex: number = 0) => {
    if (!engineRef.current) return;

    let dropX = 600;
    let dropY = 100;
    
    if (entryFunnelsRef.current.length > 0) {
      const funnel = entryFunnelsRef.current[funnelIndex % entryFunnelsRef.current.length];
      dropX = funnel.x;
      dropY = funnel.y;
    }

    const props = MARBLE_PROPERTIES[color];
    const marble = Matter.Bodies.circle(dropX, dropY, props.radius, {
      restitution: props.restitution,
      friction: props.friction,
      density: props.density,
      render: {
        sprite: {
             // If you have a marble texture, use it here. Otherwise use color.
             // texture: '/textures/marble_glass.png', 
             xScale: (props.radius * 2) / 512,
             yScale: (props.radius * 2) / 512
        },
        fillStyle: MARBLE_COLORS[color],
        strokeStyle: "#ffffff",
        lineWidth: 1,
      },
      label: `marble-${color}`,
    });

    Matter.Composite.add(engineRef.current.world, marble);
    marbleBodiesRef.current.set(marbleId, { body: marble, color, timestamp: Date.now() });
  };

  // Expose methods to global scope for React components to trigger
  useEffect(() => {
    if (isInitialized) {
      (window as any).physicsEngine = {
        dropMarble: dropMarbleInternal,
        clearAllMarbles: clearAllMarbles,
      };
    }
  }, [isInitialized, clearAllMarbles]);

  return (
    <div className="relative w-full h-full flex items-center justify-center bg-gray-900">
      <canvas
        ref={canvasRef}
        width={1200}
        height={800}
        className="border-4 border-gray-700 rounded-lg shadow-2xl"
      />
    </div>
  );
};
```
